296 class connect_message {
297 public:
298  connect_message(
299  std::uint16_t keep_alive_sec,
300  std::string const& client_id,
301  bool clean_session,
302  mqtt::optional<will> const& w,
303  mqtt::optional<std::string> const& user_name,
304  mqtt::optional<std::string> const& password
305  )
306  : fixed_header_(static_cast<char>(make_fixed_header(control_packet_type::connect, 0b0000))),
307  connect_flags_(0),
308  // protocol name length, protocol name, protocol level, connect flag, client id length, client id, keep alive
309  remaining_length_(
310  2 + // protocol name length
311  4 + // protocol name
312  1 + // protocol level
313  1 + // connect flag
314  2 + // keep alive
315  2 + // client id length
316  client_id.size() // client id
317  ),
318  protocol_name_and_level_ { 0x00, 0x04, 'M', 'Q', 'T', 'T', 0x04 },
319  client_id_(as::buffer(client_id)),
320  client_id_length_buf_{ MQTT_16BITNUM_TO_BYTE_SEQ(client_id.size()) },
321  keep_alive_buf_ { MQTT_16BITNUM_TO_BYTE_SEQ(keep_alive_sec ) }
322  {
323  utf8string_check(client_id);
324  if (clean_session) connect_flags_ |= connect_flags::clean_session;
325  if (user_name) {
326  utf8string_check(user_name.value());
327  connect_flags_ |= connect_flags::user_name_flag;
328  user_name_ = as::buffer(user_name.value());
329  add_uint16_t_to_buf(user_name_length_buf_, static_cast<std::uint16_t>(get_size(user_name_)));
330 
331  remaining_length_ += 2 + get_size(user_name_);
332  }
333  if (password) {
334  connect_flags_ |= connect_flags::password_flag;
335  password_ = as::buffer(password.value());
336  add_uint16_t_to_buf(password_length_buf_, static_cast<std::uint16_t>(get_size(password_)));
337 
338  remaining_length_ += 2 + get_size(password_);
339  }
340  if (w) {
341  connect_flags_ |= connect_flags::will_flag;
342  if (w.value().retain()) connect_flags_ |= connect_flags::will_retain;
343  connect_flags::set_will_qos(connect_flags_, w.value().qos());
344 
345  utf8string_check(w.value().topic());
346  will_topic_name_ = as::buffer(w.value().topic());
347  add_uint16_t_to_buf(
348  will_topic_name_length_buf_,
349  static_cast<std::uint16_t>(get_size(will_topic_name_))
350  );
351  if (w.value().message().size() > 0xffffL) throw will_message_length_error();
352  will_message_ = as::buffer(w.value().message());
353  add_uint16_t_to_buf(
354  will_message_length_buf_,
355  static_cast<std::uint16_t>(get_size(will_message_)));
356 
357  remaining_length_ += 2 + get_size(will_topic_name_) + 2 + get_size(will_message_);
358  }
359 
360  auto rb = remaining_bytes(remaining_length_);
361  for (auto e : rb) {
362  remaining_length_buf_.push_back(e);
363  }
364  }

502  basic_publish_message(
503  as::const_buffer const& topic_name,
504  std::uint8_t qos,
505  bool retain,
506  bool dup,
507  typename packet_id_type<PacketIdBytes>::type packet_id,

509  )

513  payload_(payload),
514  remaining_length_(publish_remaining_length(topic_name, qos, payload))
515  {
516  utf8string_check(string_view(get_pointer(topic_name), get_size(topic_name)));
517  publish::set_qos(fixed_header_, qos);
518  publish::set_retain(fixed_header_, retain);
519  publish::set_dup(fixed_header_, dup);
520 
521  auto rb = remaining_bytes(remaining_length_);
522  for (auto e : rb) {
523  remaining_length_buf_.push_back(e);
524  }
525  if (qos == qos::at_least_once ||
526  qos == qos::exactly_once) {
527  packet_id_.reserve(PacketIdBytes);
528  add_packet_id_to_buf<PacketIdBytes>::apply(packet_id_, packet_id);
529  }

532  template <typename Iterator>
533  basic_publish_message(Iterator b, Iterator e) {
534  if (b >= e) throw remaining_length_error();
535  fixed_header_ = static_cast<std::uint8_t>(*b);
536  auto qos = publish::get_qos(fixed_header_);
537  ++b;
538 
539  if (b >= e) throw remaining_length_error();
540  auto len_consumed = remaining_length(b, e);
541  remaining_length_ = std::get<0>(len_consumed);
542  auto consumed = static_cast<std::string::difference_type>(std::get<1>(len_consumed));
543 
544  std::copy(b, b + consumed, std::back_inserter(remaining_length_buf_));
545  b += consumed;
546 
547  if (b + 2 >= e) throw remaining_length_error();
548  std::copy(b, b + 2, std::back_inserter(topic_name_length_buf_));
549  auto topic_name_length = make_uint16_t(b, b + 2);
550  b += 2;
551 
552  if (b + topic_name_length >= e) throw remaining_length_error();
553  utf8string_check(string_view(&*b, topic_name_length));
554  topic_name_ = as::buffer(&*b, topic_name_length);
555  b += topic_name_length;
556 
557  switch (qos) {
558  case qos::at_most_once:
559  break;
560  case qos::at_least_once:
561  case qos::exactly_once:
562  if (b + PacketIdBytes >= e) throw remaining_length_error();
563  std::copy(b, b + PacketIdBytes, std::back_inserter(packet_id_));
564  b += PacketIdBytes;
565  break;
566  default:
567  throw protocol_error();
568  break;
569  };
570 
571  payload_ = as::buffer(&*b, static_cast<std::size_t>(std::distance(b, e)));
572  }
