299 class connect_message {
300 public:
301  connect_message(
302  std::uint16_t keep_alive_sec,
303  buffer client_id,
304  bool clean_session,
305  optional<will> w,
306  optional<buffer> user_name,
307  optional<buffer> password
308  )
309  : fixed_header_(static_cast<char>(make_fixed_header(control_packet_type::connect, 0b0000))),
310  connect_flags_(0),
311  // protocol name length, protocol name, protocol level, connect flag, client id length, client id, keep alive
312  remaining_length_(
313  2 + // protocol name length
314  4 + // protocol name
315  1 + // protocol level
316  1 + // connect flag
317  2 + // keep alive
318  2 + // client id length
319  client_id.size() // client id
320  ),
321  protocol_name_and_level_ { 0x00, 0x04, 'M', 'Q', 'T', 'T', 0x04 },
322  client_id_(force_move(client_id)),
323  client_id_length_buf_{ num_to_2bytes(boost::numeric_cast<std::uint16_t>(client_id_.size())) },
324  keep_alive_buf_ { num_to_2bytes(keep_alive_sec) }
325  {
326  utf8string_check(client_id_);
327  if (clean_session) connect_flags_ |= connect_flags::clean_session;
328  if (user_name) {
329  utf8string_check(user_name.value());
330  connect_flags_ |= connect_flags::user_name_flag;
331  user_name_ = force_move(user_name.value());
332  add_uint16_t_to_buf(user_name_length_buf_, boost::numeric_cast<std::uint16_t>(user_name_.size()));
333 
334  remaining_length_ += 2 + user_name_.size();
335  }
336  if (password) {
337  connect_flags_ |= connect_flags::password_flag;
338  password_ = force_move(password.value());
339  add_uint16_t_to_buf(password_length_buf_, boost::numeric_cast<std::uint16_t>(password_.size()));
340 
341  remaining_length_ += 2 + password_.size();
342  }
343  if (w) {
344  connect_flags_ |= connect_flags::will_flag;
345  if (w.value().retain()) connect_flags_ |= connect_flags::will_retain;
346  connect_flags::set_will_qos(connect_flags_, w.value().get_qos());
347 
348  utf8string_check(w.value().topic());
349  will_topic_name_ = force_move(w.value().topic());
350  add_uint16_t_to_buf(
351  will_topic_name_length_buf_,
352  boost::numeric_cast<std::uint16_t>(will_topic_name_.size())
353  );
354  if (w.value().message().size() > 0xffffL) throw will_message_length_error();
355  will_message_ = force_move(w.value().message());
356  add_uint16_t_to_buf(
357  will_message_length_buf_,
358  boost::numeric_cast<std::uint16_t>(will_message_.size()));
359 
360  remaining_length_ += 2 + will_topic_name_.size() + 2 + will_message_.size();
361  }
362 
363  auto rb = remaining_bytes(remaining_length_);
364  for (auto e : rb) {
365  remaining_length_buf_.push_back(e);
366  }
367  }

504 public:
505  basic_publish_message(
506  buffer topic_name,

509  bool dup,

519  + topic_name_.size() // topic name
520  + payload_.size() // payload
521  + ( (qos_value == qos::at_least_once || qos_value == qos::exactly_once)
522  ? PacketIdBytes // packet_id
523  : 0)
524  )
525  {
526  utf8string_check(topic_name_);
527  publish::set_qos(fixed_header_, qos_value);
528  publish::set_retain(fixed_header_, retain);
529  publish::set_dup(fixed_header_, dup);
530 
531  auto rb = remaining_bytes(remaining_length_);
532  for (auto e : rb) {
533  remaining_length_buf_.push_back(e);
534  }
535  if (qos_value == qos::at_least_once ||
536  qos_value == qos::exactly_once) {

542  basic_publish_message(buffer buf) {
543  if (buf.empty()) throw remaining_length_error();
544  fixed_header_ = static_cast<std::uint8_t>(buf.front());
545  qos qos_value = get_qos();
546  buf.remove_prefix(1);
547 
548  if (buf.empty()) throw remaining_length_error();
549  auto len_consumed = remaining_length(buf.begin(), buf.end());
550  remaining_length_ = std::get<0>(len_consumed);
551  auto consumed = std::get<1>(len_consumed);
552 
553  std::copy(
554  buf.begin(),
555  std::next(buf.begin(), static_cast<string_view::difference_type>(consumed)),
556  std::back_inserter(remaining_length_buf_));
557  buf.remove_prefix(consumed);
558 
559  if (buf.size() < 2) throw remaining_length_error();
560  std::copy(buf.begin(), std::next(buf.begin(), 2), std::back_inserter(topic_name_length_buf_));
561  auto topic_name_length = make_uint16_t(topic_name_length_buf_.begin(), topic_name_length_buf_.end());
562  buf.remove_prefix(2);
563 
564  if (buf.size() < topic_name_length) throw remaining_length_error();
565 
566  topic_name_ = buf.substr(0, topic_name_length);
567  utf8string_check(topic_name_);
568  buf.remove_prefix(topic_name_length);
569 
570  switch (qos_value) {
571  case qos::at_most_once:
572  break;
573  case qos::at_least_once:
574  case qos::exactly_once:
575  if (buf.size() < PacketIdBytes) throw remaining_length_error();
576  std::copy(buf.begin(), std::next(buf.begin(), PacketIdBytes), std::back_inserter(packet_id_));
577  buf.remove_prefix(PacketIdBytes);
578  break;
579  default:
580  throw protocol_error();
581  break;
582  };
583 
584  payload_ = force_move(buf);
585  }
