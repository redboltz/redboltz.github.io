424 class publish_message {
425 public:
426  publish_message(
427  as::const_buffer const& topic_name,
428  std::uint8_t qos,
429  bool retain,
430  bool dup,
431  std::uint16_t packet_id,
432  as::const_buffer const& payload
433  )
434  : fixed_header_(make_fixed_header(control_packet_type::publish, 0b0000)),
435  topic_name_(topic_name),
436  topic_name_length_buf_ { MQTT_16BITNUM_TO_BYTE_SEQ(get_size(topic_name)) },
437  payload_(payload),
438  remaining_length_(publish_remaining_length(topic_name, qos, payload))
439  {
440  publish::set_qos(fixed_header_, qos);
441  publish::set_retain(fixed_header_, retain);
442  publish::set_dup(fixed_header_, dup);
443 
444  auto rb = remaining_bytes(remaining_length_);
445  for (auto e : rb) {
446  remaining_length_buf_.push_back(e);
447  }
448  if (qos == qos::at_least_once ||
449  qos == qos::exactly_once) {
450  packet_id_.reserve(2);
451  add_uint16_t_to_buf(packet_id_, packet_id);
452  }
453  }
454 
455  template <typename Iterator>
456  publish_message(Iterator b, Iterator e) {
457  if (b >= e) throw remaining_length_error();
458  fixed_header_ = *b;
459  auto qos = publish::get_qos(fixed_header_);
460  ++b;
461 
462  if (b + 4 >= e) throw remaining_length_error();
463  auto len_consumed = remaining_length(b, b + 4);
464  remaining_length_ = std::get<0>(len_consumed);
465  auto consumed = std::get<1>(len_consumed);
466 
467  std::copy(b, b + consumed, std::back_inserter(remaining_length_buf_));
468  b += consumed;
469 
470  if (b + 2 >= e) throw remaining_length_error();
471  std::copy(b, b + 2, std::back_inserter(topic_name_length_buf_));
472  auto topic_name_length = make_uint16_t(b, b + 2);
473  b += 2;
474 
475  if (b + topic_name_length >= e) throw remaining_length_error();
476  topic_name_ = as::buffer(&*b, topic_name_length);
477  b += topic_name_length;
478 
479  switch (qos) {
480  case qos::at_most_once:
481  break;
482  case qos::at_least_once:
483  case qos::exactly_once:
484  if (b + 2 >= e) throw remaining_length_error();
485  std::copy(b, b + 2, std::back_inserter(packet_id_));
486  b += 2;
487  break;
488  default:
489  throw protocol_error();
490  break;
491  };
492 
493  payload_ = as::buffer(&*b, std::distance(b, e));
494  }
