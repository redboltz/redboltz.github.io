885  template <typename Iterator>
886  void connect_impl(Socket& socket, Iterator it, Iterator end, std::vector<v5::property_variant> props, async_handler_t func) {
887  auto self = shared_from_this();
888  as::async_connect(
889  socket.lowest_layer(), it, end,
890  [this, self, &socket, func = std::move(func), props = std::move(props)]
891  (boost::system::error_code const& ec, as::ip::tcp::resolver::iterator) mutable {
892  base::set_close_handler([this](){ handle_close(); });
893  base::set_error_handler([this](boost::system::error_code const& ec){ handle_error(ec); });
894  if (!ec) {
895  base::set_connect();
896  if (ping_duration_ms_ == 0) {
897  base::set_pre_send_handler();
898  }
899  else {
900  base::set_pre_send_handler(
901  [this] {
902  reset_timer();
903  }
904  );
905  set_timer();
906  }
907  }
908  if (base::handle_close_or_error(ec)) return;
909  handshake_socket(socket, std::move(props), std::move(func));
910  });
911  }

929  if (auto sp = wp.lock()) {
