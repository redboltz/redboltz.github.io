297 class connect_message {
298 public:
299  connect_message(
300  std::uint16_t keep_alive_sec,
301  MQTT_NS::string_view client_id,
302  bool clean_session,
303  MQTT_NS::optional<will> const& w,
304  MQTT_NS::optional<std::string> const& user_name,
305  MQTT_NS::optional<std::string> const& password
306  )
307  : fixed_header_(static_cast<char>(make_fixed_header(control_packet_type::connect, 0b0000))),
308  connect_flags_(0),
309  // protocol name length, protocol name, protocol level, connect flag, client id length, client id, keep alive
310  remaining_length_(
311  2 + // protocol name length
312  4 + // protocol name
313  1 + // protocol level
314  1 + // connect flag
315  2 + // keep alive
316  2 + // client id length
317  client_id.size() // client id
318  ),
319  protocol_name_and_level_ { 0x00, 0x04, 'M', 'Q', 'T', 'T', 0x04 },
320  client_id_(as::buffer(client_id.data(), client_id.size())),
321  client_id_length_buf_{ num_to_2bytes(boost::numeric_cast<std::uint16_t>(client_id.size())) },
322  keep_alive_buf_ { num_to_2bytes(keep_alive_sec) }
323  {
324  utf8string_check(client_id);
325  if (clean_session) connect_flags_ |= connect_flags::clean_session;
326  if (user_name) {
327  utf8string_check(user_name.value());
328  connect_flags_ |= connect_flags::user_name_flag;
329  user_name_ = as::buffer(user_name.value());
330  add_uint16_t_to_buf(user_name_length_buf_, boost::numeric_cast<std::uint16_t>(get_size(user_name_)));
331 
332  remaining_length_ += 2 + get_size(user_name_);
333  }
334  if (password) {
335  connect_flags_ |= connect_flags::password_flag;
336  password_ = as::buffer(password.value());
337  add_uint16_t_to_buf(password_length_buf_, boost::numeric_cast<std::uint16_t>(get_size(password_)));
338 
339  remaining_length_ += 2 + get_size(password_);
340  }
341  if (w) {
342  connect_flags_ |= connect_flags::will_flag;
343  if (w.value().retain()) connect_flags_ |= connect_flags::will_retain;
344  connect_flags::set_will_qos(connect_flags_, w.value().qos());
345 
346  utf8string_check(w.value().topic());
347  will_topic_name_ = as::buffer(w.value().topic());
348  add_uint16_t_to_buf(
349  will_topic_name_length_buf_,
350  boost::numeric_cast<std::uint16_t>(get_size(will_topic_name_))
351  );
352  if (w.value().message().size() > 0xffffL) throw will_message_length_error();
353  will_message_ = as::buffer(w.value().message());
354  add_uint16_t_to_buf(
355  will_message_length_buf_,
356  boost::numeric_cast<std::uint16_t>(get_size(will_message_)));
357 
358  remaining_length_ += 2 + get_size(will_topic_name_) + 2 + get_size(will_message_);
359  }
360 
361  auto rb = remaining_bytes(remaining_length_);
362  for (auto e : rb) {
363  remaining_length_buf_.push_back(e);
364  }
365  }

502 public:
503  basic_publish_message(
504  as::const_buffer const& topic_name,
505  std::uint8_t qos,
506  bool retain,
507  bool dup,
508  typename packet_id_type<PacketIdBytes>::type packet_id,
509  as::const_buffer const& payload
510  )
511  : fixed_header_(static_cast<char>(make_fixed_header(control_packet_type::publish, 0b0000))),
512  topic_name_(topic_name),
513  topic_name_length_buf_ { num_to_2bytes(boost::numeric_cast<std::uint16_t>(get_size(topic_name))) },
514  payload_(payload),
515  remaining_length_(publish_remaining_length(topic_name, qos, payload))
516  {
517  utf8string_check(string_view(get_pointer(topic_name), get_size(topic_name)));
518  publish::set_qos(fixed_header_, qos);
519  publish::set_retain(fixed_header_, retain);
520  publish::set_dup(fixed_header_, dup);
521 
522  auto rb = remaining_bytes(remaining_length_);
523  for (auto e : rb) {
524  remaining_length_buf_.push_back(e);
525  }
526  if (qos == qos::at_least_once ||
527  qos == qos::exactly_once) {
528  packet_id_.reserve(PacketIdBytes);
529  add_packet_id_to_buf<PacketIdBytes>::apply(packet_id_, packet_id);
530  }

533  basic_publish_message(MQTT_NS::string_view buf) {
534  if (buf.empty()) throw remaining_length_error();
535  fixed_header_ = static_cast<std::uint8_t>(buf.front());
536  auto qos = publish::get_qos(fixed_header_);
537  buf.remove_prefix(1);
538 
539  if (buf.empty()) throw remaining_length_error();
540  auto len_consumed = remaining_length(buf.begin(), buf.end());
541  remaining_length_ = std::get<0>(len_consumed);
542  auto consumed = std::get<1>(len_consumed);
543 
544  std::copy(
545  buf.begin(),
546  std::next(buf.begin(), static_cast<MQTT_NS::string_view::difference_type>(consumed)),
547  std::back_inserter(remaining_length_buf_));
548  buf.remove_prefix(consumed);
549 
550  if (buf.size() < 2) throw remaining_length_error();
551  std::copy(buf.begin(), std::next(buf.begin(), 2), std::back_inserter(topic_name_length_buf_));
552  auto topic_name_length = make_uint16_t(topic_name_length_buf_.begin(), topic_name_length_buf_.end());
553  buf.remove_prefix(2);
554 
555  if (buf.size() < topic_name_length) throw remaining_length_error();
556  utf8string_check(buf.substr(0, topic_name_length));
557  topic_name_ = as::buffer(buf.data(), topic_name_length);
558  buf.remove_prefix(topic_name_length);
559 
560  switch (qos) {
561  case qos::at_most_once:
562  break;
563  case qos::at_least_once:
564  case qos::exactly_once:
565  if (buf.size() < PacketIdBytes) throw remaining_length_error();
566  std::copy(buf.begin(), std::next(buf.begin(), PacketIdBytes), std::back_inserter(packet_id_));
567  buf.remove_prefix(PacketIdBytes);
568  break;
569  default:
570  throw protocol_error();
571  break;
572  };
573 
574  payload_ = as::buffer(buf.data(), buf.size());
575  }
