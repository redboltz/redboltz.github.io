869  template <typename Iterator>
870  void connect_impl(Socket& socket, Iterator it, Iterator end, std::vector<v5::property_variant> props, any session_life_keeper) {
871  as::async_connect(
872  socket.lowest_layer(), it, end,
873  [this, self = this->shared_from_this(), &socket, session_life_keeper = force_move(session_life_keeper), props = force_move(props)]
874  (boost::system::error_code const& ec, Iterator) mutable {
875  if (!ec) {
876  base::set_connect();
877  if (ping_duration_ms_ != 0) {
878  set_timer();
879  }
880  }
881  if (base::handle_close_or_error(ec)) return;
882  handshake_socket(socket, force_move(props), force_move(session_life_keeper));
883  });
884  }

904  void set_timer() {
905  tim_ping_.expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
906  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(this->shared_from_this()));
907  tim_ping_.async_wait(

909  if (auto sp = wp.lock()) {
910  sp->handle_timer(ec);
