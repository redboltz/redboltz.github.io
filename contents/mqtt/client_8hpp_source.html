475  template <typename Iterator>
476  void connect_impl(Socket& socket, Iterator it, Iterator end, async_handler_t const& func = async_handler_t()) {
477  auto self = this->shared_from_this();
478  as::async_connect(
479  socket.lowest_layer(), it, end,
480  [this, self, &socket, func]
481  (boost::system::error_code const& ec, as::ip::tcp::resolver::iterator) mutable {
482  base::set_close_handler([this](){ handle_close(); });
483  base::set_error_handler([this](boost::system::error_code const& ec){ handle_error(ec); });
484  if (!ec) {
485  base::set_connect();
486  if (ping_duration_ms_ != 0) {
487  tim_->expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
488  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(self));
489  tim_->async_wait(
490  [wp](boost::system::error_code const& ec) {
491  if (auto sp = wp.lock()) {
492  sp->handle_timer(ec);
493  }
494  }
495  );
496  }
497  }
498  if (base::handle_close_or_error(ec)) return;
499  handshake_socket(socket, func);
500  });
501  }
502 
503  void handle_timer(boost::system::error_code const& ec) {
504  if (!ec) {
505  base::pingreq();
506  tim_->expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
507  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(this->shared_from_this()));
508  tim_->async_wait(
509  [wp](boost::system::error_code const& ec) {
510  if (auto sp = wp.lock()) {
511  sp->handle_timer(ec);
512  }
513  }
514  );
515  }
516  }

518  void handle_close() {
519  if (ping_duration_ms_ != 0) tim_->cancel();
520  if (h_close_) h_close_();
521  }

523  void handle_error(boost::system::error_code const& ec) {

525  if (h_error_) h_error_(ec);
