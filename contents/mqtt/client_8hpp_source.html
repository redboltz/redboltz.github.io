210  void connect(async_handler_t const& func = async_handler_t()) {
211  as::ip::tcp::resolver r(ios_);
212  as::ip::tcp::resolver::query q(host_, port_);
213  auto it = r.resolve(q);
214  setup_socket(base::socket());
215  auto self = this->shared_from_this();
216  as::async_connect(
217  base::socket()->lowest_layer(), it,
218  [this, self, func]
219  (boost::system::error_code const& ec, as::ip::tcp::resolver::iterator) mutable {
220  base::set_close_handler([this](){ handle_close(); });
221  base::set_error_handler([this](boost::system::error_code const& ec){ handle_error(ec); });
222  if (!ec) {
223  base::set_connect();
224  if (ping_duration_ms_ != 0) {
225  tim_->expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
226  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(self));
227  tim_->async_wait(
228  [wp](boost::system::error_code const& ec) {
229  if (auto sp = wp.lock()) {
230  sp->handle_timer(ec);
231  }
232  }
233  );
234  }
235  }
236  if (base::handle_close_or_error(ec)) return;
237  handshake_socket(base::socket(), func);
238  });
239  }

466  void handle_timer(boost::system::error_code const& ec) {
467  if (!ec) {
468  base::pingreq();
469  tim_->expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
470  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(this->shared_from_this()));
471  tim_->async_wait(
472  [wp](boost::system::error_code const& ec) {
473  if (auto sp = wp.lock()) {
474  sp->handle_timer(ec);
475  }
476  }
477  );
478  }
479  }
