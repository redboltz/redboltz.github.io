827  void send_connect() {
828 
829  send_buffer sb;
830  std::size_t payload_position = 5; // Fixed Header + max size of Remaining bytes
831  sb.buf().resize(payload_position);
832  sb.buf().push_back(0x00); // Length MSB(0)
833  sb.buf().push_back(0x04); // Length LSB(4)
834  sb.buf().push_back('M');
835  sb.buf().push_back('Q');
836  sb.buf().push_back('T');
837  sb.buf().push_back('T');
838  sb.buf().push_back(0x04); // Level(4) MQTT version 3.1.1
839  std::size_t connect_flags_position = sb.buf().size();
840  char initial_value = clean_session_ ? 0b00000010 : 0;
841  sb.buf().push_back(initial_value); // Connect Flags
842  // Keep Alive MSB
843  sb.buf().push_back(static_cast<std::uint8_t>(keep_alive_sec_ >> 8));
844  // Keep Alive LSB
845  sb.buf().push_back(static_cast<std::uint8_t>(keep_alive_sec_ & 0xff));
846 
847  // client id
848  if (!utf8string::is_valid_length(client_id_)) throw utf8string_length_error();
849  if (!utf8string::is_valid_contents(client_id_)) throw utf8string_contents_error();
850  sb.buf().insert(sb.buf().size(), encoded_length(client_id_));
851  sb.buf().insert(sb.buf().size(), client_id_);
852 
853  // will
854  if (will_) {
855  char& c = sb.buf()[connect_flags_position];
856  c |= connect_flags::will_flag;
857  if (will_->retain()) c |= connect_flags::will_retain;
858  connect_flags::set_will_qos(c, will_->qos());
859 
860  if (!utf8string::is_valid_length(will_->topic())) throw utf8string_length_error();
861  if (!utf8string::is_valid_contents(will_->topic())) throw utf8string_contents_error();
862  sb.buf().insert(sb.buf().size(), encoded_length(will_->topic()));
863  sb.buf().insert(sb.buf().size(), will_->topic());
864 
865  if (will_->message().size() > 0xffff) throw will_message_length_error();
866  sb.buf().insert(sb.buf().size(), encoded_length(will_->message()));
867  sb.buf().insert(sb.buf().size(), will_->message());
868  }
869 
870  // user_name, password
871  if (user_name_) {
872  char& c = sb.buf()[connect_flags_position];
873  c |= connect_flags::user_name_flag;
874  std::string const& str = *user_name_;
875  if (!utf8string::is_valid_length(str)) throw utf8string_length_error();
876  if (!utf8string::is_valid_contents(str)) throw utf8string_contents_error();
877  sb.buf().insert(sb.buf().size(), encoded_length(str));
878  sb.buf().insert(sb.buf().size(), str);
879  }
880  if (password_) {
881  char& c = sb.buf()[connect_flags_position];
882  c |= connect_flags::password_flag;
883  std::string const& str = *password_;
884  if (str.size() > 0xffff) throw password_length_error();
885  sb.buf().insert(sb.buf().size(), encoded_length(str));
886  sb.buf().insert(sb.buf().size(), str);
887  }
888 
889  auto ptr_size = sb.finalize(make_fixed_header(control_packet_type::connect, 0));
890  as::write(*socket_, as::buffer(ptr_size.first, ptr_size.second));
891  }

925  if (buf_ & 0b10000000) {
926  if (++remaining_length_count_ > 4) throw remaining_length_error();
927  remaining_length_ += buf_ & 0b01111111;
928  as::async_read(
929  *socket_,
930  as::buffer(&buf_, 1),
931  [this](

937  handle_remaining_length();
