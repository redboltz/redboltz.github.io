857  template <typename Iterator>
858  void connect_impl(Socket& socket, Iterator it, Iterator end, std::vector<v5::property_variant> props, async_handler_t func) {
859  auto self = this->shared_from_this();
860  as::async_connect(
861  socket.lowest_layer(), it, end,
862  [this, self, &socket, func = std::move(func), props = std::move(props)]
863  (boost::system::error_code const& ec, as::ip::tcp::resolver::iterator) mutable {
864  base::set_close_handler([this](){ handle_close(); });
865  base::set_error_handler([this](boost::system::error_code const& ec){ handle_error(ec); });
866  if (!ec) {
867  base::set_connect();
868  if (ping_duration_ms_ == 0) {
869  base::set_pre_send_handler();
870  }
871  else {
872  base::set_pre_send_handler(
873  [this] {
874  reset_timer();
875  }
876  );
877  set_timer();
878  }
879  }
880  if (base::handle_close_or_error(ec)) return;
881  handshake_socket(socket, std::move(props), std::move(func));
882  });
883  }

891  void set_timer() {
892  tim_ping_.expires_from_now(boost::posix_time::milliseconds(ping_duration_ms_));
893  std::weak_ptr<this_type> wp(std::static_pointer_cast<this_type>(this->shared_from_this()));
894  tim_ping_.async_wait(
895  [wp](boost::system::error_code const& ec) {
896  if (auto sp = wp.lock()) {
897  sp->handle_timer(ec);
898  }

900  );
901  }
