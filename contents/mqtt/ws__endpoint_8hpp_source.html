88  void async_read(
89  MutableBufferSequence const& buffers,
90  ReadHandler&& handler) {

95 
96  std::shared_ptr<beast_read_handler_t> beast_read_handler;
97  if (req_size <= buffer_.size()) {
98  as::buffer_copy(buffers, buffer_.data(), req_size);
99  buffer_.consume(req_size);
100  handler(boost::system::errc::make_error_code(boost::system::errc::success), req_size);
101  return;
102  }
103 
104  beast_read_handler.reset(

106  [this, req_size, buffers, handler = std::forward<ReadHandler>(handler)]
107  (boost::system::error_code const& ec, std::shared_ptr<void> const& v) mutable {
108  if (ec) {
109  std::forward<ReadHandler>(handler)(ec, 0);
110  return;
111  }
112  if (!ws_.got_binary()) {
113  buffer_.consume(buffer_.size());
114  std::forward<ReadHandler>(handler)
115  (boost::system::errc::make_error_code(boost::system::errc::bad_message), 0);
116  return;
117  }

120  ws_.async_read(
