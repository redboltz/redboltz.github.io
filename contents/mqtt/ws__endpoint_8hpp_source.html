98  template <typename MutableBufferSequence, typename ReadHandler>
99  void async_read(
100  MutableBufferSequence const& buffers,
101  ReadHandler&& handler) {
102  auto req_size = as::buffer_size(buffers);
103 
104  using beast_read_handler_t =
105  std::function<void(boost::system::error_code const& ec, std::shared_ptr<void>)>;
106 
107  std::shared_ptr<beast_read_handler_t> beast_read_handler;
108  if (req_size <= buffer_.size()) {
109  as::buffer_copy(buffers, buffer_.data(), req_size);
110  buffer_.consume(req_size);
111  handler(boost::system::errc::make_error_code(boost::system::errc::success), req_size);
112  return;
113  }
114 
115  beast_read_handler.reset(
116  new beast_read_handler_t(
117  [this, req_size, buffers, handler = std::forward<ReadHandler>(handler)]
118  (boost::system::error_code const& ec, std::shared_ptr<void> const& v) mutable {
119  if (ec) {
120  std::forward<ReadHandler>(handler)(ec, 0);
121  return;
122  }
123  if (!ws_.got_binary()) {
124  buffer_.consume(buffer_.size());
125  std::forward<ReadHandler>(handler)
126  (boost::system::errc::make_error_code(boost::system::errc::bad_message), 0);
127  return;
128  }
129  if (req_size > buffer_.size()) {
130  auto beast_read_handler = std::static_pointer_cast<beast_read_handler_t>(v);
131  ws_.async_read(
132  buffer_,
133  as::bind_executor(
134  strand_,
135  [beast_read_handler]
136  (boost::system::error_code const& ec, std::size_t) {
137  (*beast_read_handler)(ec, beast_read_handler);
138  }
139  )
140  );
141  return;
142  }
143  as::buffer_copy(buffers, buffer_.data(), req_size);
144  buffer_.consume(req_size);

146  }
