42  strm.imbue(

56  if (auto v = boost::log::extract<severity_level>("Severity", rec)) {

62  // Shorten file path example

64  strm << boost::filesystem::path(v.get()).filename().string() << ":";
65  }
66  if (auto v = boost::log::extract<unsigned int>("MqttLine", rec)) {
67  strm << v.get() << " ";
