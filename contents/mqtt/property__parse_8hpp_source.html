240 template <typename It>
241 inline
242 std::vector<property_variant> parse(It& it, It end) {
243  std::vector<property_variant> props;
244  while (true) {
245  if (auto p = parse_one(it, end)) {
246  props.push_back(std::move(*p));
247  }
248  else {
249  break;
250  }
251  }
252  return props;
253 }

255 template <typename It>
256 inline
257 mqtt::optional<std::vector<property_variant>> parse_with_length(It& it, It end) {
258  auto r = variable_length(
259  it,
260  std::min(it + 4, end)
261  );
262  auto property_length = static_cast<typename std::iterator_traits<It>::difference_type>(std::get<0>(r));
263  it += static_cast<typename std::iterator_traits<It>::difference_type>(std::get<1>(r));
264 
265  if (end < it + property_length) {
266  return mqtt::nullopt;
267  }
268 
269  It prop_begin = it;
270  It prop_end = it + property_length;
271  auto pvs = v5::property::parse(prop_begin, prop_end);
272  std::advance(it, property_length);
273  return pvs;
274 }
