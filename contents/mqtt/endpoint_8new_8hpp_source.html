984  void handle_remaining_length(std::shared_ptr<void> const& resource) {
985  remaining_length_ += (buf_ & 0b01111111) * remaining_length_multiplier_;
986  remaining_length_multiplier_ *= 128;
987  if (remaining_length_multiplier_ > 128 * 128 * 128) throw remaining_length_error();
988  if (buf_ & 0b10000000) {
989  as::async_read(
990  *socket_,
991  as::buffer(&buf_, 1),
992  [this, resource](
993  boost::system::error_code const& ec,
994  std::size_t bytes_transferred){
995  if (handle_close_or_error(ec)) return;
996  if (bytes_transferred != 1)
997  throw read_bytes_transferred_error(1, bytes_transferred);
998  handle_remaining_length(resource);
999  });
1000  }
1001  else {
1002  payload_.resize(remaining_length_);
1003  if (remaining_length_ == 0) {
1004  handle_payload(resource);
1005  return;
1006  }
1007  as::async_read(
1008  *socket_,
1009  as::buffer(payload_),
1010  [this, resource](
1011  boost::system::error_code const& ec,
1012  std::size_t bytes_transferred){
1013  if (handle_close_or_error(ec)) return;
1014  if (bytes_transferred != remaining_length_)
1015  throw read_bytes_transferred_error(remaining_length_, bytes_transferred);
1016  handle_payload(resource);
1017  });
1018  }
1019  }

1080  void handle_connect() {
1081  std::size_t i = 0;
1082  if (remaining_length_ < 10 ||
1083  payload_[i++] != 0x00 ||
1084  payload_[i++] != 0x04 ||
1085  payload_[i++] != 'M' ||
1086  payload_[i++] != 'Q' ||
1087  payload_[i++] != 'T' ||
1088  payload_[i++] != 'T' ||
1089  payload_[i++] != 0x04) {
1090  throw protocol_error();
1091  }
1092  char byte8 = payload_[i++];
1093 
1094  std::uint16_t keep_alive;
1095  keep_alive = make_uint16_t(payload_[i], payload_[i + 1]);
1096  i += 2;
1097 
1098  if (remaining_length_ < i + 2) throw remaining_length_error();
1099 
1100  std::uint16_t client_id_length;
1101  client_id_length = make_uint16_t(payload_[i], payload_[i + 1]);
1102  i += 2;
1103  if (remaining_length_ < i + client_id_length) throw remaining_length_error();
1104 
1105  std::string client_id(payload_.data() + i, client_id_length);
1106  i += client_id_length;
1107 
1108  bool clean_session = connect_flags::has_clean_session(byte8);
1109  boost::optional<will> w;
1110  if (connect_flags::has_will_flag(byte8)) {
1111  std::uint16_t topic_name_length;
1112  topic_name_length = make_uint16_t(payload_[i], payload_[i + 1]);
1113  i += 2;
1114  if (remaining_length_ < i + topic_name_length) throw remaining_length_error();
1115  std::string topic_name(payload_.data() + i, topic_name_length);
1116  i += topic_name_length;
1117  std::uint16_t will_message_length;
1118  will_message_length = make_uint16_t(payload_[i], payload_[i + 1]);
1119  i += 2;
1120  if (remaining_length_ < i + will_message_length) throw remaining_length_error();
1121  std::string will_message(payload_.data() + i, topic_name_length);
1122  i += will_message_length;
1123  w = will(topic_name,
1124  will_message,
1125  connect_flags::has_will_retain(byte8),
1126  connect_flags::will_qos(byte8));
1127  }
1128  boost::optional<std::string> user_name;
1129  if (connect_flags::has_user_name_flag(byte8)) {

1325  void send_connect(std::uint16_t keep_alive_sec) {
1326 
1327  send_buffer sb;
1328  std::size_t payload_position = 5; // Fixed Header + max size of Remaining bytes
1329  sb.buf()->resize(payload_position);
1330  sb.buf()->push_back(0x00); // Length MSB(0)
1331  sb.buf()->push_back(0x04); // Length LSB(4)
1332  sb.buf()->push_back('M');
1333  sb.buf()->push_back('Q');
1334  sb.buf()->push_back('T');
1335  sb.buf()->push_back('T');
1336  sb.buf()->push_back(0x04); // Level(4) MQTT version 3.1.1
1337  std::size_t connect_flags_position = sb.buf()->size();
1338  char initial_value = clean_session_ ? 0b00000010 : 0;
1339  sb.buf()->push_back(initial_value); // Connect Flags
1340  // Keep Alive MSB
1341  sb.buf()->push_back(static_cast<std::uint8_t>(keep_alive_sec >> 8));
1342  // Keep Alive LSB
1343  sb.buf()->push_back(static_cast<std::uint8_t>(keep_alive_sec & 0xff));
1344 
1345  // endpoint id
1346  if (!utf8string::is_valid_length(client_id_)) throw utf8string_length_error();
1347  if (!utf8string::is_valid_contents(client_id_)) throw utf8string_contents_error();
1348  sb.buf()->insert(sb.buf()->size(), encoded_length(client_id_));
1349  sb.buf()->insert(sb.buf()->size(), client_id_);
1350 
1351  // will
1352  if (will_) {
1353  char& c = (*sb.buf())[connect_flags_position];
1354  c |= connect_flags::will_flag;
1355  if (will_->retain()) c |= connect_flags::will_retain;
1356  connect_flags::set_will_qos(c, will_->qos());
1357 
1358  if (!utf8string::is_valid_length(will_->topic())) throw utf8string_length_error();
1359  if (!utf8string::is_valid_contents(will_->topic())) throw utf8string_contents_error();
1360  sb.buf()->insert(sb.buf()->size(), encoded_length(will_->topic()));
1361  sb.buf()->insert(sb.buf()->size(), will_->topic());
1362 
1363  if (will_->message().size() > 0xffff) throw will_message_length_error();
1364  sb.buf()->insert(sb.buf()->size(), encoded_length(will_->message()));
1365  sb.buf()->insert(sb.buf()->size(), will_->message());
1366  }
1367 
1368  // user_name, password
1369  if (user_name_) {
1370  char& c = (*sb.buf())[connect_flags_position];
1371  c |= connect_flags::user_name_flag;
1372  std::string const& str = *user_name_;
1373  if (!utf8string::is_valid_length(str)) throw utf8string_length_error();
1374  if (!utf8string::is_valid_contents(str)) throw utf8string_contents_error();
1375  sb.buf()->insert(sb.buf()->size(), encoded_length(str));
1376  sb.buf()->insert(sb.buf()->size(), str);
1377  }
1378  if (password_) {
1379  char& c = (*sb.buf())[connect_flags_position];
1380  c |= connect_flags::password_flag;
1381  std::string const& str = *password_;
1382  if (str.size() > 0xffff) throw password_length_error();
1383  sb.buf()->insert(sb.buf()->size(), encoded_length(str));
1384  sb.buf()->insert(sb.buf()->size(), str);
1385  }
1386 
1387  auto ptr_size = sb.finalize(make_fixed_header(control_packet_type::connect, 0));
1388  as::write(*socket_, as::buffer(ptr_size.first, ptr_size.second));
1389  }
