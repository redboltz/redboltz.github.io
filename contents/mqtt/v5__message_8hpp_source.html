608  basic_publish_message(Iterator b, Iterator e) {
609  if (b >= e) throw remaining_length_error();
610  fixed_header_ = static_cast<std::uint8_t>(*b);
611  auto qos = publish::get_qos(fixed_header_);
612  ++b;
613 
614  if (b >= e) throw remaining_length_error();
615  auto len_consumed = remaining_length(b, e);
616  remaining_length_ = std::get<0>(len_consumed);
617  auto consumed = std::get<1>(len_consumed);
618 
619  std::copy(b, b + static_cast<typename std::iterator_traits<Iterator>::difference_type>(consumed), std::back_inserter(remaining_length_buf_));
620  b += static_cast<typename std::iterator_traits<Iterator>::difference_type>(consumed);
621 
622  if (b + 2 >= e) throw remaining_length_error();
623  std::copy(b, b + 2, std::back_inserter(topic_name_length_buf_));
624  auto topic_name_length = make_uint16_t(b, b + 2);
625  b += 2;
626 
627  if (b + topic_name_length >= e) throw remaining_length_error();
628  utf8string_check(string_view(&*b, topic_name_length));
629  topic_name_ = as::buffer(&*b, topic_name_length);
630  b += topic_name_length;
631 
632  switch (qos) {
633  case qos::at_most_once:
634  break;
635  case qos::at_least_once:
636  case qos::exactly_once:
637  if (b + PacketIdBytes >= e) throw remaining_length_error();
638  std::copy(b, b + PacketIdBytes, std::back_inserter(packet_id_));
639  b += PacketIdBytes;
640  break;
641  default:
